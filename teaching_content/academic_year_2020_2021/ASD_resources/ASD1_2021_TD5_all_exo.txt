#include<stdio.h> // include stdio.h library
#define MAX 50

int main(void)
{    
    int T[MAX], min, max,i,j,n = 0;

    printf("Enter size of the array .max 50: ");
    scanf("%d", &n);
 
    for(i=0; i<n; i++)
    {
        printf("Enter element [%d] : ",i+1);
        scanf("%d",&T[i]);
    }

    min = max = T[0]; // assign the first element to max and min
    // min = max = T[0] = 5
    // T[1] = 10 i = 1 max = T[1] = 10 min = 5
    // T[2]  = 3 i = 2 min = t[2] = 3 max = 10
    for(int i = 0; i < n; i++)
    {
        if(T[i] < min)
        {
            min = T[i];
        }

        if(T[i] > max)
        {
            max = T[i];
        }
    }        

    printf("Min = %d\n", min);
    printf("Max = %d\n", max);

    return 0; // return 0 to operating system
}

#include <stdio.h>

#define MAXROW		10
#define MAXCOL		5

int main()
{
	int matrix[MAXROW][MAXCOL];
	int i,j,r,c;
	float sum,prod;
	float moy;
	
	printf("Enter number of Rows :");
	scanf("%d",&r);
	printf("Enter number of Cols :");
	scanf("%d",&c);

	printf("\nEnter matrix elements :\n");
	for(i=0;i< r;i++)
	{
		for(j=0;j< c;j++)
		{
			printf("Enter element [%d,%d] : ",i+1,j+1);
			scanf("%d",&matrix[i][j]);
		}
	}
	printf("\n");
	/*sum of all elements*/
    sum=0;		/*initializing sum*/
	prod = 1;

	for(i=0;i< r;i++)
	{
		for(j=0;j< c;j++)
		{
			printf("%d\t",matrix[i][j]);	/*print elements*/
			sum		+=	matrix[i][j];
			prod	*=	matrix[i][j];
		}	
        printf("\n");
	}
	moy = sum/(r * c);
    printf("SUM : %.2f\n",sum);
	printf("Product : %.2f\n",prod);
	printf("Moy : %.2f\n",moy);

}
